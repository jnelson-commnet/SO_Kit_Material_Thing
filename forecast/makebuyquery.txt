Select PART.NUM as PARTNUM, CUSTOMSET.INFO as "Make/Buy", PARTCOST.AVGCOST
from CUSTOMSET, PART left join PARTCOST
on PART.ID = PARTCOST.PARTID
-- where PART.ACTIVEFLAG = 1 AND PART.ID = CUSTOMSET.RECORDID AND CUSTOMFIELDID = 64
where PART.ID = CUSTOMSET.RECORDID AND CUSTOMFIELDID = 64
order by PART.NUM
