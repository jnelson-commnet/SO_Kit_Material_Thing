select PART.NUM as PARTNUM, SUM(QTYINVENTORYTOTALS.QTYONHAND - QTYINVENTORYTOTALS.QTYNOTAVAILABLE) as QTYONHAND, SUM(QTYINVENTORYTOTALS.QTYALLOCATED) as QTYALLOCATED, SUM(QTYINVENTORYTOTALS.QTYONORDER) as QTYONORDER
from PART, QTYINVENTORYTOTALS
where QTYINVENTORYTOTALS.PARTID = PART.ID
group by PART.NUM
